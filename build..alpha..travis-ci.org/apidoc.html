    function is_http_iri(value, allowHttps) {
    if (!is_iri(value)) {
        return;
    }

    var splitted = [];
    var scheme = '';
    var authority = '';
    var path = '';
    var port = '';
    var query = '';
    var fragment = '';
    var out = '';

    // from RFC 3986
    splitted = splitUri(value);
    scheme = splitted[1];
    authority = splitted[2];
    path = splitted[3];
    query = splitted[4];
    fragment = splitted[5];

    if (!scheme)  return;

    if(allowHttps) {
        if (scheme.toLowerCase() != 'https') return;
    } else {
        if (scheme.toLowerCase() != 'http') return;
    }

    // fully-qualified URIs must have an authority section that is
    // a valid host
    if (!authority) {
        return;
    }

    // enable port component
    if (/:(\d+)$/.test(authority)) {
        port = authority.match(/:(\d+)$/)[0];
        authority = authority.replace(/:\d+$/, '');
    }

    out += scheme + ':';
    out += '//' + authority;

    if (port) {
        out += port;
    }

    out += path;

    if(query && query.length){
        out += '?' + query;
    }

    if(fragment && fragment.length){
        out += '#' + fragment;
    }

    return out;
}

    var authority = '';
    var path = '';
    var query = '';
    var fragment = '';
